
Start Program
WHEN page finishes loading (DOMContentLoaded)
    Get all required DOM elements:
        wordDisplay
        keyboard
        remainingGuessesEl
        gameMessageEl
        resetBtn
        hangmanParts (head, body, arms, legs, face)

    Define game variables:
        selectedWord
        correctLetters = empty list
        wrongLetters = empty list
        remainingGuesses = 6
        gameOver = false

    Define wordCategories dictionary:
        animals → list of animal words
        countries → list of country words
        fruits → list of fruit words

    CALL initGame()
END

# Initialize Game
FUNCTION initGame()

    Reset correctLetters to empty
    Reset wrongLetters to empty
    Set remainingGuesses to 6
    Set gameOver to false
    Clear gameMessage

    Select random category from wordCategories
    Select random word from that category
    Store in selectedWord

    Display selected category on screen
    Display remaining guesses

    Hide all hangman body parts

    Clear wordDisplay area

    FOR each letter in selectedWord
        Create empty letter box
        Store correct letter in data attribute
        Add box to wordDisplay
    END FOR

    Clear keyboard area

    FOR letters A to Z
        Create button for each letter
        Add click event → call handleGuess(letter)
        Add button to keyboard
    END FOR

END FUNCTION

# Handle Letter Guess
FUNCTION handleGuess(letter)

    IF gameOver is true
        RETURN

    IF letter already in correctLetters OR wrongLetters
        RETURN

    IF selectedWord contains letter THEN
        Add letter to correctLetters
        CALL updateWordDisplay()

        Mark keyboard button as correct and used

        IF checkWin() is true THEN
            Set gameOver to true
            Display "Congratulations! You won!"
        END IF

    ELSE
        Add letter to wrongLetters
        Decrease remainingGuesses by 1
        Update remainingGuesses display

        Mark keyboard button as wrong and used

        CALL updateHangmanDrawing()

        IF remainingGuesses equals 0 THEN
            Set gameOver to true
            Display "Game Over" message
            Show correct word
            Show hangman face
            Reveal all letters
        END IF
    END IF

END FUNCTION

# Update Hangman Drawing
FUNCTION updateHangmanDrawing()

    SWITCH number of wrongLetters

        CASE 1 → show head
        CASE 2 → show body
        CASE 3 → show left arm
        CASE 4 → show right arm
        CASE 5 → show left leg
        CASE 6 → show right leg

    END SWITCH

END FUNCTION

# Update Word Display
FUNCTION updateWordDisplay()

    FOR each letter box in wordDisplay
        IF letter is in correctLetters
            Show the letter
        END IF
    END FOR

END FUNCTION

# Check Win Condition
FUNCTION checkWin()

    FOR each letter in selectedWord
        IF letter NOT in correctLetters
            RETURN false
    END FOR

    RETURN true

END FUNCTION

# Keyboard Input (Physical Keyboard)
WHEN user presses a key
    IF key is a letter (A–Z)
        Convert to uppercase
        CALL handleGuess(letter)
END

# Reset Button
WHEN reset button is clicked
    CALL initGame()
END
